118 template <class T>
119 vcg::Shot<T> track2ShotCPU(vcg::Shot<T> &refCamera, vcg::Trackball *track)
120 {
121  vcg::Shot<T> view;
122 
123  double _near, _far;
124  _near = 0.1;
125  _far = 100;
126 
127  //get shot extrinsics matrix
128  vcg::Matrix44<T> shotExtr;
129  refCamera.GetWorldToExtrinsicsMatrix().ToMatrix(shotExtr);
130 
131  vcg::Matrix44<T> model2;
132  model2 = (shotExtr)* vcg::Matrix44<T>::Construct(track->Matrix());
133  vcg::Matrix44<T> model;
134  model2.ToMatrix(model);
135 
136  //get translation out of modelview
137  vcg::Point3<T> tra;
138  tra[0] = model[0][3]; tra[1] = model[1][3]; tra[2] = model[2][3];
139  model[0][3] = model[1][3] = model[2][3] = 0;
140 
141  //get pure rotation out of modelview
142  double det = model.Determinant();
143  double idet = 1/pow(det, 1/3.0); //inverse of the determinant
144  model *= idet;
145  model[3][3] = 1;
146  view.Extrinsics.SetRot(model);
147 
148  //get pure translation out of modelview
149  vcg::Matrix44<T> imodel = model;
150  vcg::Transpose(imodel);
151  tra = -(imodel*tra);
152  tra *= idet;
153  view.Extrinsics.SetTra(vcg::Point3<T>::Construct(tra));
154 
155  //use same current intrinsics
156  view.Intrinsics = refCamera.Intrinsics;
157 
158  return view;
159 }

167  void shot2Track(const vcg::Shot<T> &from, const float cameraDist, vcg::Trackball &tb)
168 {
169  vcg::Quaternion<T> qfrom; qfrom.FromMatrix(from.Extrinsics.Rot());
170  tb.track.rot = vcg::Quaternionf::Construct(qfrom);
171  tb.track.tra = (vcg::Point3f::Construct(-from.Extrinsics.Tra()));
172  tb.track.tra += vcg::Point3f::Construct(tb.track.rot.Inverse().Rotate(vcg::Point3f(0,0,cameraDist)))*(1/tb.track.sca);
173 }
